      Prand : ListPattern : Pattern : AbstractFunction : Object

      embed values randomly chosen from a list
      Source: ListPatterns.sc
      See also: Pxrand, Pwrand

      Description

      Embed one item from the list at random for each repeat.

SynthDef(\help_sinegrain,
    { arg out=0, freq=440, sustain=0.05;
        var env;
        env = EnvGen.kr(Env.perc(0.01, sustain, 0.2), doneAction: Done.freeSelf);
        Out.ar(out, SinOsc.ar(freq, 0, env))
    }).add;
)

a = Prand(#[60, 61, 63, 65, 72], inf).asStream;
Routine({
    loop({
        Synth(\help_sinegrain, [\freq, a.next.midicps]);
        0.2.wait;
    })
}).play;
)
