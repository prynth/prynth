        Description

        x' = s(y - x)
y' = x(r - z) - y
z' = xy - bz

        Examples

        { LorenzTrig.ar(11025, 44100) }.play(s);

// randomly modulate params
(
{ LorenzTrig.ar(
    11025, 44100,
    LFNoise0.kr(1, 2, 10),
    20,
    LFNoise0.kr(1, 1.5, 2),
    0.02
) * 0.2 }.play(s);
)

// as a frequency control
{ SinOsc.ar(Decay.ar(LorenzTrig.ar(1, 8, h:0.02))*800+900)*0.4 }.play(s);
